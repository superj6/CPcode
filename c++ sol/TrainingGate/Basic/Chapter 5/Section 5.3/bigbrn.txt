#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;

int main() {
	ifstream fin ("bigbrn.in");
	ofstream fout ("bigbrn.out");
	
	int n, t;
	fin >> n >> t;
	
	int graph[n][n];
	
	for(int i = 0; i < n; i++) for(int j = 0; j < n; j++) graph[i][j] = -1;
	
	for(int i = 0; i < t; i++){
		int a, b;
		fin >> a >> b;
		a--;
		b--;
		
		graph[a][b] = 0;
	}
	
	int maxv = 0;
	
	for(int i = 0; i < n; i++){
		for(int j = 0; j < n; j++){
			if(graph[i][j] == 0){
				graph[i][j] == 0;
			}else if(i == 0 || j == 0){
				graph[i][j] = 1;
			}else{
				graph[i][j] = 1 + min(graph[i-1][j], min(graph[i][j-1], graph[i-1][j-1]));
			}
			
			maxv = max(maxv, graph[i][j]);
		}
	}
	
	fout << maxv << endl;
	
	return 0;
}
